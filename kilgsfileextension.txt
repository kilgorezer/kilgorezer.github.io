var fileSystem = {}

class kilg {
  getInfo() {
    return {
      id: 'fvnxjnkvnjnknjvknjjvcnjnjvnjxnjkvnjkjkn',
      name: 'File System Manager (Backslash Enforced)',
      blocks: [
        {
          opcode: 'hello',
          blockType: Scratch.BlockType.REPORTER,
          text: 'Get the file in the location C:\\[ONE]',
          arguments: {
            ONE: {
                type: Scratch.ArgumentType.STRING
            }
          }
        }
      ]
    };
  }

  hello(args) {
    args = args.ONE.toString().split("\\")
    var tempfsys = Object.create(fileSystem)
    for(var i = 0; i < args.length; i++) {
      tempfsys = fileSystem[args[i]]
    }
    if(typeof(tempfsys) != "Object") {
      return tempfsys
    }
  }
  store(args) {
    //
  }
}

Scratch.extensions.register(new kilg());
